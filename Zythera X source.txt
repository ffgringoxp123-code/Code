function convertStringToTable(P)
	local Q = {};
	for P in string.gmatch(P, "([^,]+)") do
		local L = P:match("^%s*(.-)%s*$");
		table.insert(Q, L);
	end;
	return Q;
end;

function u.render(Q, P)
	local L = { top_left = Vector2.new(), top_right = Vector2.new(), bottom_right = Vector2.new() };
	local function D(P, Q)
		L.top_left = Q;
		L.top_right = Q + Vector2.new(P.X, 0);
		L.bottom_right = Q + P;
	end;
	local function k()
		local D = L.top_left;
		local k = L.top_right;
		local n = L.bottom_right;
		local r = x:viewport_point_to_world(D, P);
		local t = x:viewport_point_to_world(k, P);
		local i = x:viewport_point_to_world(n, P);
		local X = (t - r).Magnitude;
		local E = (t - i).Magnitude;
		if not Q._root then
			return;
		end;
		Q._root.CFrame = CFrame.fromMatrix((r + i) / 2, workspace.CurrentCamera.CFrame.XVector, workspace.CurrentCamera.CFrame.YVector, workspace.CurrentCamera.CFrame.ZVector);
		Q._root.Mesh.Scale = Vector3.new(X, E, 0);
	end;
	local function n()
		local P = x:get_offset();
		local L = Q._frame.AbsoluteSize - Vector2.new(P, P);
		local n = Q._frame.AbsolutePosition + Vector2.new(P / 2, P / 2);
		D(L, n);
		task.spawn(k);
	end;
	l.cframe_update = (workspace.CurrentCamera:GetPropertyChangedSignal("CFrame")):Connect(k);
	l.viewport_size_update = (workspace.CurrentCamera:GetPropertyChangedSignal("ViewportSize")):Connect(k);
	l.field_of_view_update = (workspace.CurrentCamera:GetPropertyChangedSignal("FieldOfView")):Connect(k);
	l.frame_absolute_position = (Q._frame:GetPropertyChangedSignal("AbsolutePosition")):Connect(n);
	l.frame_absolute_size = (Q._frame:GetPropertyChangedSignal("AbsoluteSize")):Connect(n);
	task.spawn(k);
end;

local function og(P, Q, L)
	return P + (Q - P) * L;
end;
local function wg(P, Q, L, D)
	return og(og(P, Q, D), og(Q, L, D), D);
end;
local function pg(P, Q)
	local L = (P + Q) * .5;
	local D = P - Q;
	if D.Magnitude < 5 then
		return Q;
	end;
	local k = math.atan2(D.Z, D.X);
	local n = D.Magnitude * .7;
	local r = L + Vector3.new(math.cos(k + math.pi / 2), 0, math.sin(k + math.pi / 2)) * n;
	local t = L + Vector3.new(math.cos(k - math.pi / 2), 0, math.sin(k - math.pi / 2)) * n;
	return (r - L):Dot(P - L) < 0 and r or t;
end;
local function Wg(P, Q, L)
	dg = dg + L;
	local D = math.clamp(dg / .9, 0, 1);
	if D >= 1 then
		if (P - Q).Magnitude >= 10 then
			dg = 0;
		end;
		sg = nil;
		return Q;
	end;
	if not sg then
		sg = pg(P, Q);
	end;
	return wg(P, sg, Q, D);
end;
