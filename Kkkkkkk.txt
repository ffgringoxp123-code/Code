local curveMethod = "camera"

local function ExecuteParry()
    local camera = workspace.CurrentCamera
    local cameraCFrame = camera and camera.CFrame or CFrame.new()

    if curveMethod == "upwards" then
        local look = cameraCFrame.LookVector
        local upDir = Vector3.new(look.X, 10, look.Z).Unit
        cameraCFrame = CFrame.new(cameraCFrame.Position, cameraCFrame.Position + upDir)
    elseif curveMethod == "backwards" then
        cameraCFrame = CFrame.lookAt(cameraCFrame.Position, cameraCFrame.Position - cameraCFrame.LookVector)
    elseif curveMethod == "random" then
        local randomDir = Vector3.new(
            math.random(-1000, 1000) * 0.001,
            math.random(-350, 1000) * 0.001,
            math.random(-1000, 1000) * 0.001
        ).Unit
        cameraCFrame = CFrame.new(cameraCFrame.Position, cameraCFrame.Position + randomDir)
    end

    if next(ParryRemotes) then
        for remote, originalArgs in pairs(ParryRemotes) do
            local modifiedArgs = {
                originalArgs[1],
                originalArgs[2],
                0,
                cameraCFrame,
                originalArgs[5],
                originalArgs[6]
            }
            if remote:IsA("RemoteEvent") then
                remote:FireServer(unpack(modifiedArgs))
            elseif remote:IsA("RemoteFunction") then
                remote:InvokeServer(unpack(modifiedArgs))
            end
        end
    else
        VIM:SendMouseButtonEvent(0, 0, 0, true, game, 0)
        VIM:SendMouseButtonEvent(0, 0, 0, false, game, 0)
    end

    if Parry_Animation_API then
        pcall(function()
            if type(Parry_Animation_API) == "function" then
                Parry_Animation_API()
            elseif type(Parry_Animation_API) == "table" and Parry_Animation_API.Play then
                Parry_Animation_API:Play()
            end
        end)
    end
    return true
end

local parryTimes = {}
local parryCount = 0
local parryHead = 1
local ballPrevVelocity = {}
local antiCurveData = {}

local CURVE_BOOST_FACTOR = 1.3
local CURVE_ACCEL_THRESHOLD = 60
local SMOOTH_ALPHA = 0.3
local CONSISTENT_FRAMES_NEEDED = 3

function recordAutoParry()
    local now = tick()
    if parryCount < 5 then
        parryCount = parryCount + 1
    else
        parryHead = (parryHead % 5) + 1
    end
    parryTimes[((parryHead + parryCount - 2) % 5) + 1] = now
end

local function countRecent()
    local now = tick()
    local count = 0
    for i = 1, parryCount do
        local idx = (parryHead + i - 2) % 5 + 1
        if now - parryTimes[idx] < 0.5 then
            count = count + 1
        end
    end
    return count
end

local function getAntiCurveMultiplier(ball, playerPos, currentVelocity)
    local prev = ballPrevVelocity[ball]
    if not prev then
        ballPrevVelocity[ball] = currentVelocity
        antiCurveData[ball] = { smoothAx = 0, smoothAy = 0, smoothAz = 0, curveFrames = 0 }
        return 1.0
    end

    local ax = currentVelocity.X - prev.X
    local ay = currentVelocity.Y - prev.Y
    local az = currentVelocity.Z - prev.Z
    ballPrevVelocity[ball] = currentVelocity

    local data = antiCurveData[ball]
    if not data then
        data = { smoothAx = 0, smoothAy = 0, smoothAz = 0, curveFrames = 0 }
        antiCurveData[ball] = data
    end

    data.smoothAx = data.smoothAx * (1 - SMOOTH_ALPHA) + ax * SMOOTH_ALPHA
    data.smoothAy = data.smoothAy * (1 - SMOOTH_ALPHA) + ay * SMOOTH_ALPHA
    data.smoothAz = data.smoothAz * (1 - SMOOTH_ALPHA) + az * SMOOTH_ALPHA

    local accelMag = math.sqrt(data.smoothAx*data.smoothAx + data.smoothAy*data.smoothAy + data.smoothAz*data.smoothAz)
    if accelMag < 0.1 then
        data.curveFrames = 0
        return 1.0
    end

    local dx = playerPos.X - ball.Position.X
    local dy = playerPos.Y - ball.Position.Y
    local dz = playerPos.Z - ball.Position.Z
    local dist = math.sqrt(dx*dx + dy*dy + dz*dz)
    if dist < 0.01 then
        data.curveFrames = 0
        return 1.0
    end

    local unitToPlayerX = dx / dist
    local unitToPlayerY = dy / dist
    local unitToPlayerZ = dz / dist

    local radialAccel = data.smoothAx*unitToPlayerX + data.smoothAy*unitToPlayerY + data.smoothAz*unitToPlayerZ
    local lateralAccelSq = accelMag*accelMag - radialAccel*radialAccel
    if lateralAccelSq < 0 then lateralAccelSq = 0 end
    local lateralAccel = math.sqrt(lateralAccelSq)

    if lateralAccel > CURVE_ACCEL_THRESHOLD and radialAccel > 0 then
        data.curveFrames = data.curveFrames + 1
        if data.curveFrames >= CONSISTENT_FRAMES_NEEDED then
            local severity = math.min(lateralAccel / (CURVE_ACCEL_THRESHOLD * 2), 1.0)
            return 1.0 + (CURVE_BOOST_FACTOR - 1.0) * severity
        end
    else
        data.curveFrames = 0
    end
    return 1.0
end

local function checkAutoSpam()
    if not autoSpamEnabled then
        if aggressiveMode then
            aggressiveMode = false
            StopAutoSpam()
        end
        return
    end

    local recent = countRecent()
    local shouldBeAggressive = (recent >= 3)

    if shouldBeAggressive ~= aggressiveMode then
        aggressiveMode = shouldBeAggressive
        if aggressiveMode then
            StartAutoSpam()
        else
            StopAutoSpam()
        end
    end
end

local basePreClick = 1.4
local maxHorizontalMiss = 5

local function AutoParryLoop()
    local character = player.Character
    local rootPart = character and character:FindFirstChild("HumanoidRootPart")
    if not rootPart then return end

    local playerPos = rootPart.Position
    local px, py, pz = playerPos.X, playerPos.Y, playerPos.Z
    local pingMs = statsPingItem:GetValue()
    local pingFactor = math.clamp(pingMs * 0.01, 1, 20)

    for _, ball in ipairs(BallsFolder:GetChildren()) do
        if not ball:GetAttribute("realBall") then continue end
        if ball:GetAttribute("target") ~= player.Name then continue end

        local state = ballStates[ball]
        if not state then InitBallState(ball); state = ballStates[ball] end
        if state.parried then continue end

        local zoomies = ball:FindFirstChild("zoomies")
        if not zoomies then continue end
        local velVec = zoomies.VectorVelocity
        local vx, vy, vz = velVec.X, velVec.Y, velVec.Z

        local bx, by, bz = ball.Position.X, ball.Position.Y, ball.Position.Z
        local dx, dy, dz = px - bx, py - by, pz - bz
        local distSq = dx*dx + dy*dy + dz*dz
        if distSq < 1 then continue end
        local dist = math.sqrt(distSq)

        local approachSpeed = (dx*vx + dy*vy + dz*vz) / dist
        if approachSpeed <= 0 then continue end

        local horizontalSpeedSq = vx*vx + vz*vz
        if horizontalSpeedSq > 0.001 then
            local horizontalMiss = math.abs(dx*vz - dz*vx) / math.sqrt(horizontalSpeedSq)
            if horizontalMiss > maxHorizontalMiss then continue end
        end

        local cappedSpeed = math.min(approachSpeed, 650)
        local baseRadius = 2.4 + cappedSpeed * 0.002
        local parryRadius = pingFactor + math.max(approachSpeed / (baseRadius * ACCURACY_FACTOR), 9.5)
        local curveMultiplier = getAntiCurveMultiplier(ball, playerPos, velVec)
        parryRadius = parryRadius * basePreClick * curveMultiplier

        if dist <= parryRadius then
            state.parried = true
            ExecuteParry()
            recordAutoParry()
        end
    end

    checkAutoSpam()
end

local function SetAutoParryConnection()
    if autoParryConnection then
        autoParryConnection:Disconnect()
        autoParryConnection = nil
    end
    if autoParryEnabled then
        autoParryConnection = RunService.Stepped:Connect(AutoParryLoop)
    end
end

local function ToggleAutoParry(on)
    autoParryEnabled = on
    SetAutoParryConnection()
    if not on then
        StopAutoSpam()
        aggressiveMode = false
    end
end

local Visuals = {
    Enabled = false,
    HitEffects = false,
    Tracers = false,
    Outline = false,
    VisualParts = {},
    TracerLines = {},
    OutlineHandles = {},
    Color = Color3.fromRGB(255, 255, 255)

local function FetchBalls()
    local balls = {}
    for _, b in ipairs(BallsFolder:GetChildren()) do
        if b:GetAttribute("realBall") then
            table.insert(balls, b)
        end
    end
    return balls
end

local function UpdateVisualizer()
    if not Visuals.Enabled then
        for _, part in pairs(Visuals.VisualParts) do
            pcall(function() part:Destroy() end)
        end
        Visuals.VisualParts = {}
        return
    end
    for _, ball in ipairs(FetchBalls()) do
        if ball and ball.Parent then
            local speed = ball.AssemblyLinearVelocity.Magnitude
            local size = math.clamp(speed / 5, 3.5, 500)
            if not Visuals.VisualParts[ball] then
                local sphere = Instance.new("Part")
                sphere.Name = "VisualPart"
                sphere.Anchored = true
                sphere.CanCollide = false
                sphere.CastShadow = false
                sphere.Shape = Enum.PartType.Ball
                sphere.Transparency = 0
                sphere.Material = Enum.Material.ForceField
                sphere.Color = Visuals.Color
                sphere.Parent = ball
                Visuals.VisualParts[ball] = sphere
            end
            Visuals.VisualParts[ball].Position = ball.Position
            Visuals.VisualParts[ball].Size = Vector3.new(size, size, size)
            Visuals.VisualParts[ball].Color = Visuals.Color
        end
    end
    for ball, part in pairs(Visuals.VisualParts) do
        if not ball or not ball.Parent then
            pcall(function() part:Destroy() end)
            Visuals.VisualParts[ball] = nil
        end
    end
end

local function UpdateTracers()
    if not Visuals.Tracers then
        for _, line in pairs(Visuals.TracerLines) do
            pcall(function() line:Destroy() end)
        end
        Visuals.TracerLines = {}
        return
    end
    local root = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
    if not root then return end
    for _, ball in ipairs(FetchBalls()) do
        if ball and ball.Parent then
            if not Visuals.TracerLines[ball] then
                local line = Instance.new("Part")
                line.Name = "Tracer"
                line.Anchored = true
                line.CanCollide = false
                line.CastShadow = false
                line.Material = Enum.Material.Neon
                line.Size = Vector3.new(0.2, 0.2, 0)
                line.Color = Visuals.Color
                line.Parent = workspace
                Visuals.TracerLines[ball] = line
            end
            local start = ball.Position
            local finish = root.Position
            local dir = (finish - start).Unit
            local dist = (finish - start).Magnitude
            local line = Visuals.TracerLines[ball]
            line.Size = Vector3.new(0.2, 0.2, dist)
            line.CFrame = CFrame.new(start + dir * dist / 2, finish)
            line.Color = Visuals.Color
        end
    end
    for ball, line in pairs(Visuals.TracerLines) do
        if not ball or not ball.Parent then
            pcall(function() line:Destroy() end)
            Visuals.TracerLines[ball] = nil
        end
    end
end

local function UpdateOutline()
    if not Visuals.Outline then
        for _, handle in pairs(Visuals.OutlineHandles) do
            pcall(function() handle:Destroy() end)
        end
        Visuals.OutlineHandles = {}
        return
    end
    for _, ball in ipairs(FetchBalls()) do
        if ball and ball.Parent then
            if not Visuals.OutlineHandles[ball] then
                local hl = Instance.new("Highlight", ball)
                hl.FillTransparency = 1
                hl.OutlineTransparency = 0
                hl.OutlineColor = Visuals.Color
                Visuals.OutlineHandles[ball] = hl

                Visuals.OutlineHandles[ball].OutlineColor = Visuals.Color
            end
        end
    end
    for ball, handle in pairs(Visuals.OutlineHandles) do
        if not ball or not ball.Parent then
            pcall(function() handle:Destroy() end)
            Visuals.OutlineHandles[ball] = nil
        end
    end
end

                statsLabel.Text = "waiting..."
                return
            end
            local maxSpeed = 0
            for _, b in ipairs(balls) do
                local zoom = b:FindFirstChild("zoomies")
                if zoom then
                    local spd = zoom.VectorVelocity.Magnitude
                    if spd > maxSpeed then maxSpeed = spd end
                end

            local txt = string.format("ball velocity: %.0f", maxSpeed)
            if maxSpeed >= 2200 then txt = txt .. " (LIMIT!)" end
            statsLabel.Text = txt
            statsLabel.TextColor3 = maxSpeed >= 2200 and Color3.new(1, 0, 0) or Color3.new(1, 1, 1)
        end
        local conn = RunService.Heartbeat:Connect(update)
        statsGui.Destroying:Connect(function()
            conn:Disconnect()
        end)
    else
        if statsGui then statsGui:Destroy(); statsGui = nil end
    end
end

        pcall(function() bodyColors.LeftArmColor3 = desc.LeftArmColor end)
        pcall(function() bodyColors.RightArmColor3 = desc.RightArmColor end)
        pcall(function() bodyColors.LeftLegColor3 = desc.LeftLegColor end)
        pcall(function() bodyColors.RightLegColor3 = desc.RightLegColor end)
    end
end
local avatarConnection = nil
local function ToggleAvatarChanger(on)
    if on and avatarName ~= "" then
        player:ClearCharacterAppearance()
        ApplyAvatar(player.Character, avatarName)
        avatarConnection = player.CharacterAdded:Connect(function(char)
            task.wait(0.65)
            if avatarChangerEnabled and avatarName ~= "" then
                player:ClearCharacterAppearance()
                task.wait(0.1)
                ApplyAvatar(char, avatarName)
            end
        end)
    else
        if avatarConnection then
            avatarConnection:Disconnect()
            avatarConnection = nil
        end
        if player.Character then
            player:ClearCharacterAppearance()
            ApplyAvatar(player.Character, player.Name)
        end
    end
end

local Music = { Active = false, Volume = 0.5, CurrentId = "9126639173", Sound = nil }
local function PlayMusic()
    if Music.Sound then pcall(function() Music.Sound:Stop() end) end
    if Music.Active then
        local sound = Instance.new("Sound")
        sound.SoundId = "rbxassetid://" .. Music.CurrentId
        sound.Volume = Music.Volume
        sound.Looped = true
        sound.Parent = player:WaitForChild("PlayerGui")
        pcall(function() sound:Play() end)
        Music.Sound = sound
    end
end

local manualSpamButton = nil
local manualSpamButtonGui = nil

local function UpdateManualSpamButtonUI()
    if not manualSpamButton then return end
    if manualSpamRunning then
        manualSpamButton.Text = "ON"
    else
        manualSpamButton.Text = "OFF"
    end
end

local function CreateManualSpamButton()
    if manualSpamButtonGui then manualSpamButtonGui:Destroy() end
    manualSpamButtonGui = Instance.new("ScreenGui")
    manualSpamButtonGui.Name = "ManualSpamButton"
    manualSpamButtonGui.ResetOnSpawn = false
    manualSpamButtonGui.Parent = player:WaitForChild("PlayerGui")
    
    manualSpamButton = Instance.new("TextButton")
    manualSpamButton.Size = UDim2.new(0, 160, 0, 90)
    manualSpamButton.Position = UDim2.new(0.5, -80, 0.8, -45)
    manualSpamButton.BackgroundColor3 = Color3.new(0, 0, 0)
    manualSpamButton.BackgroundTransparency = 0.2
    manualSpamButton.Text = "OFF"
    manualSpamButton.TextColor3 = Color3.new(1, 1, 1)
    manualSpamButton.TextSize = 24
    manualSpamButton.Font = Enum.Font.GothamBold
    manualSpamButton.BorderSizePixel = 0
    manualSpamButton.Parent = manualSpamButtonGui
    
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 12)
    corner.Parent = manualSpamButton
    
    local dragging = false
    local dragStartPos = nil
    local buttonStartPos = nil
    
    local function startDrag(input)
        dragging = true
        dragStartPos = input.Position
        buttonStartPos = manualSpamButton.Position
    end
    
    local function updateDrag(input)
        if not dragging then return end
        local delta = input.Position - dragStartPos
        manualSpamButton.Position = UDim2.new(
            buttonStartPos.X.Scale,
            buttonStartPos.X.Offset + delta.X,
            buttonStartPos.Y.Scale,
            buttonStartPos.Y.Offset + delta.Y
        )
    end
    
    local function endDrag()
        dragging = false
    end
    
    manualSpamButton.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            startDrag(input)
        end
    end)
    
    manualSpamButton.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            updateDrag(input)
        end
    end)
    
    manualSpamButton.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            endDrag()
        end
    end)
    
    manualSpamButton.MouseButton1Click:Connect(function()
        if manualSpamRunning then
            StopManualSpam()
        else
            StartManualSpam()
        end
        UpdateManualSpamButtonUI()
        if Toggles and Toggles.ManualSpam then
            Toggles.ManualSpam:SetValue(manualSpamRunning)
        end
    end)
end

local function DestroyManualSpamButton()
    if manualSpamButtonGui then
        manualSpamButtonGui:Destroy()
        manualSpamButtonGui = nil
        manualSpamButton = nil
    end
end
